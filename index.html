        function sendAIMessage() {
            const input = document.getElementById('chat-input');
            const message = input.value.trim();
            if (!message) return;
            
            const messagesContainer = document.getElementById('chat-messages');
            const userDiv = document.createElement('div');
            userDiv.className = 'flex gap-2 justify-end';
            userDiv.innerHTML = `
${message}
👤
`;
            messagesContainer.appendChild(userDiv);
            input.value = '';
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
            
            const typingDiv = document.createElement('div');
            typingDiv.className = 'flex gap-2';
            typingDiv.innerHTML = `
AI
●●●
`;
            messagesContainer.appendChild(typingDiv);
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
            
            setTimeout(() => {
                typingDiv.remove();
                let response = 'أنا هنا لمساعدتك! يمكنني بناء مواقع كاملة بالذكاء الاصطناعي. فقط اكتب وصفاً مفصلاً للموقع الذي تريده في الصندوق الموجود في الصفحة الرئيسية.';
                const aiDiv = document.createElement('div');
                aiDiv.className = 'flex gap-2';
                aiDiv.innerHTML = `
AI
${response}
`;
                messagesContainer.appendChild(aiDiv);
                messagesContainer.scrollTop = messagesContainer.scrollHeight;
            }, 1000);
        }
        
        document.getElementById('chat-input')?.addEventListener('keypress', (e) => { if (e.key === 'Enter') sendAIMessage(); });
